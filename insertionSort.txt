 void insertionSort(int a[],int n)
 {
 	int key;
 	for(i=1;i<n;i++)
 	{
 		key=a[i];
 		j=i-1;
 		while(j>=0 && a[j]>key)
 		{
 			a[j+1]=a[j];
 			j=j-1;
		 }
		  a[j+1]=key;
	 }
 }

 void display(int a[], int n) 
 {
	for ( i = 0; i < n;i++) 
	{
	printf("%d  ", a[i]);
	}
	printf("\n");
 }
